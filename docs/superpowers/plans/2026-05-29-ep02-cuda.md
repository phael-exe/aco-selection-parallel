**Tech Stack:** C++17, CUDA (nvcc), cuRAND, `cudaEvent_t` para timing, Makefile existente (target `cuda`).

```cpp
#pragma once
#include <cuda_runtime.h>
#include <cstdio>
#include <cstdlib>

#define CUDA_CHECK(call)                                                    \
    do {                                                                    \
        cudaError_t err = (call);                                           \
        if (err != cudaSuccess) {                                           \
            fprintf(stderr, "CUDA error %s:%d: %s\n",                      \
                    __FILE__, __LINE__, cudaGetErrorString(err));           \
            exit(EXIT_FAILURE);                                             \
        }                                                                   \
    } while (0)

struct TransferTiming {
    float h2d_ms;
    float d2h_ms;
};

struct GpuBuffers {
    double* d_X;          // [N * F] features
    double* d_dist;       // [N * N] distâncias (NULL se on-the-fly)
    double* d_vis;        // [N * N] visibilidades (NULL se on-the-fly)
    double* d_pheromone;  // [N] vetor de feromônio 1D
    int*    d_colony;     // [N * N] colônia (solução)
    int     N;
    int     F;
    bool    onthefly;     // true se N > 10000
};

inline GpuBuffers alloc_gpu(int N, int F, bool onthefly) {
    GpuBuffers buf;
    buf.N = N; buf.F = F; buf.onthefly = onthefly;
    CUDA_CHECK(cudaMalloc(&buf.d_X,         (size_t)N * F * sizeof(double)));
    CUDA_CHECK(cudaMalloc(&buf.d_pheromone, (size_t)N     * sizeof(double)));
    CUDA_CHECK(cudaMalloc(&buf.d_colony,    (size_t)N * N * sizeof(int)));
    if (!onthefly) {
        CUDA_CHECK(cudaMalloc(&buf.d_dist, (size_t)N * N * sizeof(double)));
        CUDA_CHECK(cudaMalloc(&buf.d_vis,  (size_t)N * N * sizeof(double)));
    } else {
        buf.d_dist = nullptr;
        buf.d_vis  = nullptr;
    }
    return buf;
}

inline void free_gpu(GpuBuffers& buf) {
    CUDA_CHECK(cudaFree(buf.d_X));
    CUDA_CHECK(cudaFree(buf.d_pheromone));
    CUDA_CHECK(cudaFree(buf.d_colony));
    if (buf.d_dist) CUDA_CHECK(cudaFree(buf.d_dist));
    if (buf.d_vis)  CUDA_CHECK(cudaFree(buf.d_vis));
}

inline TransferTiming upload_X(GpuBuffers& buf, const double* h_X) {
    cudaEvent_t t0, t1;
    CUDA_CHECK(cudaEventCreate(&t0));
    CUDA_CHECK(cudaEventCreate(&t1));
    CUDA_CHECK(cudaEventRecord(t0));
    CUDA_CHECK(cudaMemcpy(buf.d_X, h_X,
                          (size_t)buf.N * buf.F * sizeof(double),
                          cudaMemcpyHostToDevice));
    CUDA_CHECK(cudaEventRecord(t1));
    CUDA_CHECK(cudaEventSynchronize(t1));
    TransferTiming t; t.d2h_ms = 0.0f;
    CUDA_CHECK(cudaEventElapsedTime(&t.h2d_ms, t0, t1));
    CUDA_CHECK(cudaEventDestroy(t0));
    CUDA_CHECK(cudaEventDestroy(t1));
    return t;
}

inline float download_colony(const GpuBuffers& buf, int* h_colony) {
    cudaEvent_t t0, t1;
    CUDA_CHECK(cudaEventCreate(&t0));
    CUDA_CHECK(cudaEventCreate(&t1));
    CUDA_CHECK(cudaEventRecord(t0));
    CUDA_CHECK(cudaMemcpy(h_colony, buf.d_colony,
                          (size_t)buf.N * buf.N * sizeof(int),
                          cudaMemcpyDeviceToHost));
    CUDA_CHECK(cudaEventRecord(t1));
    CUDA_CHECK(cudaEventSynchronize(t1));
    float ms;
    CUDA_CHECK(cudaEventElapsedTime(&ms, t0, t1));
    CUDA_CHECK(cudaEventDestroy(t0));
    CUDA_CHECK(cudaEventDestroy(t1));
    return ms;
}
```

#pragma once
#include <cuda_runtime.h>
#include <curand_kernel.h>

// Atomic add para double (compatível com GPUs < sm_60 via CAS)
__device__ inline double atomicAddDouble(double* addr, double val) {
#if __CUDA_ARCH__ >= 600
    return atomicAdd(addr, val);
#else
    unsigned long long int* addr_ull = (unsigned long long int*)addr;
    unsigned long long int old = *addr_ull, assumed;
    do {
        assumed = old;
        old = atomicCAS(addr_ull, assumed,
            __double_as_longlong(val + __longlong_as_double(assumed)));
    } while (assumed != old);
    return __longlong_as_double(old);
#endif
}

// T12: Distância euclidiana par-a-par (apenas para N <= 10k)

```cpp
#include "kernels.cuh"
#include <math.h>

// T12 — Distância euclidiana par-a-par
// Cada thread calcula dist[i][j] e dist[j][i] (explorar simetria)
// Grid: blockDim=256, gridDim=ceil(N*N/2 / 256) — upper triangle only
__global__ void pairwise_distance_kernel(
    const double* X, double* dist, int N, int F)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * (N - 1) / 2;  // upper triangle, sem diagonal
    if (idx >= total) return;

    // Converter índice linear em (i, j) do triângulo superior
    // Fórmula: linha i onde triangular_number(i) > idx
    int i = (int)((-1.0 + sqrt(1.0 + 8.0 * idx)) / 2.0);
    int j = idx - i * (i + 1) / 2;
    // Ajuste: trabalhar em triângulo inferior para garantir j < i+1
    // Recalcular com esquema padrão: percorre (row, col) com row > col
    // idx mapeia para: row = 1..N-1, col = 0..row-1
    // Para robustez usar mapeamento simples row-by-row:
    int row = (int)((1.0 + sqrt(1.0 + 8.0 * (double)idx)) / 2.0);
    int col = idx - row * (row - 1) / 2;
    i = row; j = col;

    if (i >= N || j >= i) return;

    double sum = 0.0;
    for (int k = 0; k < F; ++k) {
        double diff = X[i * F + k] - X[j * F + k];
        sum += diff * diff;
    }
    double d = sqrt(sum);
    dist[i * N + j] = d;
    dist[j * N + i] = d;
    if (i == j) dist[i * N + j] = 0.0;
}
```

// T13 — Visibilidade η[i][j] = 1/dist[i][j]
// Cada thread processa um elemento da matriz N×N
__global__ void visibility_kernel(
    const double* dist, double* vis, int N)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= N * N) return;
    int i = idx / N;
    int j = idx % N;
    if (i == j) {
        vis[idx] = 0.0;
    } else {
        double d = dist[idx];
        vis[idx] = (d == 0.0) ? 0.0 : 1.0 / d;
    }
}

    int i = blockIdx.x * blockDim.x + threadIdx.x;

    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    pheromone[i] += deposit[i];
}

- [ ] **Step 1: Adicionar `init_curand_kernel` e `ant_construction_kernel`**

Este é o kernel mais complexo. Cada thread (formiga) constrói sua solução iterativamente usando roulette wheel estocástico. O feromônio é vetor 1D[N].

```cpp
// Inicialização do estado cuRAND por thread
__global__ void init_curand_kernel(curandState* states, unsigned long long seed, int N) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    if (tid >= N) return;
    curand_init(seed, tid, 0, &states[tid]);
}

// T14 + T15 — Construção de solução + depósito de feromônio
// colony[ant * N + j] = 1 se a formiga `ant` visitou instância j, 0 se rejeitada, -1 se não visitada
// Cada formiga começa em colony[ant*N + ant] = 1 (instância inicial)
__global__ void ant_construction_kernel(
    const double* X,
    const double* dist,
    const double* vis,
    const double* pheromone,
    int*          colony,
    double*       deposit,
    curandState*  rng_states,
    int N, int F, bool onthefly)
{
    int ant = blockIdx.x * blockDim.x + threadIdx.x;
    if (ant >= N) return;

    curandState local_state = rng_states[ant];

    // Posição atual da formiga: começa na sua própria instância
    int pos = ant;

    // Loop: enquanto existir instância não visitada (-1) nessa formiga
    bool has_unvisited = true;
    while (has_unvisited) {
        // Calcular soma total de atração (smell) para instâncias não visitadas
        double smell = 0.0;
        for (int j = 0; j < N; ++j) {
            if (colony[ant * N + j] < 0) {
                double eta;
                if (onthefly) {
                    // calcular visibilidade inline
                    double sum = 0.0;
                    for (int k = 0; k < F; ++k) {
                        double d = X[pos * F + k] - X[j * F + k];
                        sum += d * d;
                    }
                    double dist_ij = sqrt(sum);
                    eta = (dist_ij == 0.0) ? 0.0 : 1.0 / dist_ij;
                } else {
                    eta = vis[pos * N + j];
                }
                smell += pheromone[j] * eta;
            }
        }

        has_unvisited = false;
        for (int j = 0; j < N; ++j) {
            if (colony[ant * N + j] < 0) {
                has_unvisited = true;
                double eta;
                if (onthefly) {
                    double sum = 0.0;
                    for (int k = 0; k < F; ++k) {
                        double d = X[pos * F + k] - X[j * F + k];
                        sum += d * d;
                    }
                    double dist_ij = sqrt(sum);
                    eta = (dist_ij == 0.0) ? 0.0 : 1.0 / dist_ij;
                } else {
                    eta = vis[pos * N + j];
                }

                double prob = (smell == 0.0) ? 0.0 : (pheromone[j] * eta) / smell;
                int ajk = (curand_uniform(&local_state) < 0.5f) ? 0 : 1;
                double final_prob = prob * ajk;

                if (final_prob != 0.0) {
                    // Aceitar: mover para j
                    double tour_contrib = (onthefly) ? 0.0 : dist[pos * N + j];
                    if (!onthefly && tour_contrib > 0.0) {
                        // Depósito proporcional (Q=1): aproximação simples
                        // O depósito exato requer tour_length completo — acumulamos ao final
                    }
                    colony[ant * N + j] = 1;
                    pos = j;
                    break;
                } else {
                    colony[ant * N + j] = 0;
                }
            }
        }
    }

    // Calcular tour_length para depósito: soma distâncias percorridas
    // Para simplificar: tour_length = número de instâncias visitadas (proxy)
    // Depósito: 1/N por instância visitada (Q=1 / N)
    double tour_len = 0.0;
    for (int j = 0; j < N; ++j) {
        if (colony[ant * N + j] == 1 && j != ant) {
            if (!onthefly && dist != nullptr) {
                tour_len += dist[ant * N + j];
            } else {
                tour_len += 1.0;
            }
        }
    }

    double ant_deposit = (tour_len == 0.0) ? 0.0 : 1.0 / tour_len;
    for (int j = 0; j < N; ++j) {
        if (colony[ant * N + j] == 1 && j != ant) {
            atomicAddDouble(&deposit[j], ant_deposit);
        }
    }

    rng_states[ant] = local_state;
}
```

// T19 — 1-NN paralelo para avaliação de qualidade
// Cada thread avalia uma instância de teste: encontra vizinho mais próximo em X_train (selecionado)
// selected_mask[j] = 1 se instância j foi selecionada pela colônia
// predictions[t] = classe do vizinho mais próximo da instância de teste t
__global__ void knn_1nn_kernel(
    const double* X_train, const int* selected_mask,
    const double* X_test,  int* predictions,
    int N, int F, int N_test, int n_classes)
{
    int t = blockIdx.x * blockDim.x + threadIdx.x;
    if (t >= N_test) return;

    double best_dist = 1e300;
    int    best_label = 0;

    // X_train tem as features nas primeiras F-1 colunas, classe na última
    for (int j = 0; j < N; ++j) {
        if (selected_mask[j] != 1) continue;
        double sum = 0.0;
        for (int k = 0; k < F - 1; ++k) {
            double d = X_test[t * F + k] - X_train[j * F + k];
            sum += d * d;
        }
        if (sum < best_dist) {
            best_dist = sum;
            best_label = (int)X_train[j * F + (F - 1)];
        }
    }
    predictions[t] = best_label;
}
```

using namespace std;

// Lê CSV separado por ';', retorna vetor de linhas (sem header)
// e número de colunas (features + classe)
vector<vector<double>> read_csv(const string& path, int& F_out) {
    ifstream file(path);
    if (!file) { cerr << "Erro ao abrir: " << path << "\n"; exit(1); }
    vector<vector<double>> rows;
    string line;
    bool first = true;
    while (getline(file, line)) {
        if (first) { // skip header
            // Contar colunas
            stringstream ss(line);
            string cell;
            F_out = 0;
            while (getline(ss, cell, ';')) F_out++;
            first = false;
            continue;
        }
        vector<double> row;
        stringstream ss(line);
        string cell;
        while (getline(ss, cell, ';')) {
            try { row.push_back(stod(cell)); }
            catch (...) { row.push_back(0.0); }
        }
        if ((int)row.size() == F_out)
            rows.push_back(row);
    }
    return rows;
}

void write_solutions_csv(const vector<int>& colony, int N, const string& path) {
    ofstream out(path);
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            out << colony[i * N + j];
            if (j + 1 < N) out << ";";
        }
        out << "\n";
    }
}

int main(int argc, char** argv) {
    if (argc < 2) {
        cerr << "Uso: " << argv[0] << " <dataset_name> [data_dir]\n";
        cerr << "Ex:  " << argv[0] << " heart_failure data/baseline\n";
        return 1;
    }

    string dataset  = argv[argc - 1];
    string data_dir = (argc >= 3) ? argv[1] : "data/baseline";
    string path     = data_dir + "/" + dataset + ".csv";

    // Parâmetros ACO
    const double initial_pheromone = 1.0;
    const double evaporation_rate  = 0.1;
    const double Q                 = 1.0;
    const int    BLOCK             = 256;

    // Leitura do dataset
    int F = 0;
    auto rows = read_csv(path, F);
    int N = (int)rows.size();
    cout << "Dataset: " << path << " | N=" << N << " F=" << F << "\n";

    // Montar array flat [N * F]
    vector<double> h_X(N * F);
    for (int i = 0; i < N; ++i)
        for (int j = 0; j < F; ++j)
            h_X[i * F + j] = rows[i][j];

    // Estratégia de memória
    bool onthefly = (N > 10000);
    cout << "Estrategia: " << (onthefly ? "on-the-fly" : "pre-computado") << "\n";

    // Alocar GPU
    GpuBuffers buf = alloc_gpu(N, F, onthefly);

    // Upload X com medição de tempo
    cudaEvent_t ev0, ev1; float ms;
    cudaEventCreate(&ev0); cudaEventCreate(&ev1);
    cudaEventRecord(ev0);
    cudaMemcpy(buf.d_X, h_X.data(), (size_t)N * F * sizeof(double), cudaMemcpyHostToDevice);
    cudaEventRecord(ev1); cudaEventSynchronize(ev1);
    cudaEventElapsedTime(&ms, ev0, ev1);
    printf("Transferencia H->D X: %.2f ms\n", ms);

    // Inicializar feromônio
    vector<double> h_phe(N, initial_pheromone);
    cudaMemcpy(buf.d_pheromone, h_phe.data(), N * sizeof(double), cudaMemcpyHostToDevice);

    // Inicializar colônia: -1 exceto diagonal = 1
    vector<int> h_colony(N * N, -1);
    for (int i = 0; i < N; ++i) h_colony[i * N + i] = 1;
    cudaMemcpy(buf.d_colony, h_colony.data(), (size_t)N * N * sizeof(int), cudaMemcpyHostToDevice);

    // Pré-computar distâncias e visibilidades (se não onthefly)
    if (!onthefly) {
        int total_pairs = N * (N - 1) / 2;
        int grid_dist   = (total_pairs + BLOCK - 1) / BLOCK;
        pairwise_distance_kernel<<<grid_dist, BLOCK>>>(buf.d_X, buf.d_dist, N, F);
        cudaDeviceSynchronize();

        int grid_vis = (N * N + BLOCK - 1) / BLOCK;
        visibility_kernel<<<grid_vis, BLOCK>>>(buf.d_dist, buf.d_vis, N, F);
        cudaDeviceSynchronize();
    }

    // Inicializar cuRAND
    curandState* d_rng;
    cudaMalloc(&d_rng, N * sizeof(curandState));
    int grid_rng = (N + BLOCK - 1) / BLOCK;
    init_curand_kernel<<<grid_rng, BLOCK>>>(d_rng, 42ULL, N);
    cudaDeviceSynchronize();

    // Buffer de depósito de feromônio
    double* d_deposit;
    cudaMalloc(&d_deposit, N * sizeof(double));

    // Loop principal do ACO — iterar até colônia completa
    int grid_ants = (N + BLOCK - 1) / BLOCK;
    int grid_phe  = (N + BLOCK - 1) / BLOCK;

    cudaEventRecord(ev0);

    bool colony_complete = false;
    int  max_iters = N * 10;  // safety bound
    int  iter      = 0;

    while (!colony_complete && iter < max_iters) {
        // Zerar buffer de depósito
        cudaMemset(d_deposit, 0, N * sizeof(double));

        // Construção de soluções + depósito
        ant_construction_kernel<<<grid_ants, BLOCK>>>(
            buf.d_X, buf.d_dist, buf.d_vis,
            buf.d_pheromone, buf.d_colony,
            d_deposit, d_rng,
            N, F, onthefly);
        cudaDeviceSynchronize();

        // Aplicar depósito
        apply_deposit_kernel<<<grid_phe, BLOCK>>>(
            buf.d_pheromone, d_deposit, N, Q);

        // Evaporação
        pheromone_evaporation_kernel<<<grid_phe, BLOCK>>>(
            buf.d_pheromone, N, evaporation_rate);
        cudaDeviceSynchronize();

        // Verificar se a colônia está completa (nenhum -1 restante)
        cudaMemcpy(h_colony.data(), buf.d_colony,
                   (size_t)N * N * sizeof(int), cudaMemcpyDeviceToHost);
        colony_complete = true;
        for (int x : h_colony) {
            if (x < 0) { colony_complete = false; break; }
        }
        ++iter;
    }

    cudaEventRecord(ev1); cudaEventSynchronize(ev1);
    cudaEventElapsedTime(&ms, ev0, ev1);
    printf("Tempo compute GPU: %.2f ms (%d iters)\n", ms, iter);

    // Download resultado final
    cudaEventRecord(ev0);
    cudaMemcpy(h_colony.data(), buf.d_colony,
               (size_t)N * N * sizeof(int), cudaMemcpyDeviceToHost);
    cudaEventRecord(ev1); cudaEventSynchronize(ev1);
    cudaEventElapsedTime(&ms, ev0, ev1);
    printf("Transferencia D->H colony: %.2f ms\n", ms);

    // Salvar resultado
    write_solutions_csv(h_colony, N, "results/solutions_cuda.csv");
    cout << "Resultado salvo em results/solutions_cuda.csv\n";

    // Cleanup
    free_gpu(buf);
    cudaFree(d_rng);
    cudaFree(d_deposit);
    cudaEventDestroy(ev0); cudaEventDestroy(ev1);

    return 0;
}
```
